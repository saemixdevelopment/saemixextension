# Parameters
psi1<-c(1.5, 5, 0.04)
omega1<-diag(c(0.05, 0.0125, 0.05))
res1<-c(0.1,0.1)

# Model
model1cpt<-function(psi,id,xidep) {
  tim<-xidep[,1]
  dose<-xidep[,2]
  ka<-psi[id,1]
  V<-psi[id,2]
  CL<-psi[id,3]
  k<-CL/V
  ypred<-dose*ka/(V*(ka-k))*(exp(-k*tim)-exp(-ka*tim))
  return(ypred)
}
